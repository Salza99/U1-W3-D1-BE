esercizio 1
SELECT * from cliente
WHERE EXTRACT (YEAR FROM data_di_nascita) = 1982

esercizio 2
SELECT COUNT(*) FROM fatture
WHERE iva = 20

esercizio 3
SELECT data_fattura, COUNT(*) as numero_di_fatture, SUM(importo) as somma_importi_fatture FROM fatture
GROUP BY data_fattura

esercizio 4
SELECT * FROM prodotti
WHERE data_attivazione = 2017 AND (in_produzione = true OR in_commercio = true)

esercizio 5
SELECT data_fattura, COUNT(*) as numero_fatture FROM fatture
WHERE iva = 20
GROUP BY data_fattura

esercizio 6
SELECT data_fattura FROM fatture
WHERE tipologia = 'A'
GROUP BY data_fatture
HAVING COUNT(*) > 2

esercizio 7
SELECT numero_fattura, importo, iva, data_fattura, denominazione FROM fatture
WHERE fornitori IN(SELECT denominazione FROM fornitori)
